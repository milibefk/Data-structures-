#include <iostream>

void bubbleSort(int arr[], int n) {
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                // Swap arr[j] and arr[j + 1]
                int temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}

int main() {
    int arr_1[] = {40, 5, 6, 0, 1, 3};
    int n = sizeof(arr_1) / sizeof(arr_1[0]);

    bubbleSort(arr_1, n);

    std::cout << "Bubble Sort Result: ";
    for (int i = 0; i < n; i++) {
        std::cout << arr_1[i] << " ";
    }

    return 0;
}
